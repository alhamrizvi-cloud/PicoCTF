when we are extracting .tar file we get another .tar file, so i made a python script using claude 

```
#!/usr/bin/env python3
import tarfile
import os
import glob

def extract_nested_tars(starting_tar):
    current_tar = starting_tar
    count = 0

    while current_tar and os.path.exists(current_tar):
        print(f"Extracting: {current_tar}")
        
        with tarfile.open(current_tar) as tar:
            tar.extractall()
        
        count += 1
        os.remove(current_tar)

        # Find the next .tar file extracted
        tar_files = glob.glob("*.tar")
        current_tar = tar_files[0] if tar_files else None

    print(f"\nDone! Extracted {count} tar files.")

if __name__ == "__main__":
    tar_files = glob.glob("*.tar")
    
    if not tar_files:
        print("No .tar files found in current directory.")
    else:
        # Start with the highest-numbered tar (e.g. 1000.tar)
        starting = sorted(tar_files, reverse=True)[0]
        print(f"Starting with: {starting}\n")
        extract_nested_tars(starting)
```
